258#define macro_isMatrixUnitary(m, dim, retVal) { \
259 /* elemRe_ and elemIm_ must not exist in caller scope */ \
260 qreal elemRe_, elemIm_; \
261 retVal = 1; \
262 /* check m * ConjugateTranspose(m) == Identity */ \
263 for (int r=0; r < (dim); r++) { \
264 for (int c=0; c < (dim); c++) { \
265 /* m[r][...] * ConjugateTranspose(m)[...][c] */ \
266 elemRe_ = 0; \
267 elemIm_ = 0; \
268 for (int i=0; i < (dim); i++) { \
269 /* m[r][i] * conj(m[c][i]) */ \
270 elemRe_ += m.real[r][i]*m.real[c][i] + m.imag[r][i]*m.imag[c][i]; \
271 elemIm_ += m.imag[r][i]*m.real[c][i] - m.real[r][i]*m.imag[c][i]; \
272 } \
273 /* check distance from identity */ \
274 if ((absReal(elemIm_) > REAL_EPS) || \
275 (r == c && absReal(elemRe_ - 1) > REAL_EPS) || \
276 (r != c && absReal(elemRe_ ) > REAL_EPS)) { \
277 retVal = 0; \
278 break; \
279 } \
280 } \
281 if (retVal == 0) \
282 break; \
283 } \
284}

304#define macro_isCompletelyPositiveMap(ops, numOps, opDim) { \
305 for (int r=0; r<(opDim); r++) { \
306 for (int c=0; c<(opDim); c++) { \
307 qreal elemRe_ = 0; \
308 qreal elemIm_ = 0; \
309 for (int n=0; n<(numOps); n++) { \
310 for (int k=0; k<(opDim); k++) { \
311 elemRe_ += ops[n].real[k][r]*ops[n].real[k][c] + ops[n].imag[k][r]*ops[n].imag[k][c]; \
312 elemIm_ += ops[n].real[k][r]*ops[n].imag[k][c] - ops[n].imag[k][r]*ops[n].real[k][c]; \
313 } \
314 } \
315 qreal dist_ = absReal(elemIm_) + absReal(elemRe_ - ((r==c)? 1:0)); \
316 if (dist_ > REAL_EPS) \
317 return 0; \
318 } \
319 } \
320 return 1; \
321}

337int areUniqueQubits(int* qubits, int numQubits) {
338 long long int mask = 0;
339 long long int bit;
340 for (int q=0; q < numQubits; q++) {
341 bit = 1LL << qubits[q];
342 if (mask & bit)
343 return 0;
344 mask |= bit;
345 }
346 return 1;
347}

357void validateNumRanks(int numRanks, const char* caller) {
358
359 /* silly but robust way to determine if numRanks is a power of 2,
360 * in lieu of bit-twiddling (e.g. AND with all-ones) which may be
361 * system / precsision dependent
362 */
363 int isValid = 0;
364 for (int exp2 = 1; exp2 <= numRanks; exp2 *= 2)
365 if (exp2 == numRanks)
366 isValid = 1;
367
368 QuESTAssert(isValid, E_INVALID_NUM_RANKS, caller);
369}

371void validateNumQubitsInQureg(int numQubits, int numRanks, const char* caller) {
372 QuESTAssert(numQubits>0, E_INVALID_NUM_CREATE_QUBITS, caller);
373
374 // mustn't be more amplitudes than can fit in the type
375 unsigned int maxQubits = calcLog2(SIZE_MAX);
376 QuESTAssert( numQubits <= maxQubits, E_NUM_AMPS_EXCEED_TYPE, caller);
377
378 // must be at least one amplitude per node
379 long unsigned int numAmps = (1UL<<numQubits);
380 QuESTAssert(numAmps >= numRanks, E_DISTRIB_QUREG_TOO_SMALL, caller);
381}

831void validatePhaseFuncTerms(int numQubits, enum bitEncoding encoding, qreal* coeffs, qreal* exponents, int numTerms, long long int* overrideInds, int numOverrides, const char* caller) {
832 QuESTAssert(numTerms>0, E_INVALID_NUM_PHASE_FUNC_TERMS, caller);
833
834 int hasFractionExpo = 0;
835 int hasNegativeExpo = 0;
836 for (int t=0; t<numTerms; t++) {
837 // this is only true if exponent is precisely an integer, else pow() will NaN
838 if (floor(exponents[t]) != exponents[t])
839 hasFractionExpo = 1;
840 if (exponents[t] < 0)
841 hasNegativeExpo = 1;
842 }
843
844 // ensure negative exponents are supplied along with a zero-index override
845 if (hasNegativeExpo) {
846 int hasZeroOverride = 0;
847 for (int v=0; v<numOverrides; v++) {
848 if (overrideInds[v] == 0LL) {
849 hasZeroOverride = 1;
850 break;
851 }
852 }
853 QuESTAssert(hasZeroOverride, E_NEGATIVE_EXPONENT_WITHOUT_ZERO_OVERRIDE, caller);
854 }
855
856 // ensure fractional powers not used with negative numbers, unless overriden
857 if (hasFractionExpo && encoding == TWOS_COMPLEMENT) {
858 long long int numNegInds = (1LL << (numQubits-1));
859
860 // immediately disqualify if insufficient overrides are given to cover every negative number
861 QuESTAssert(numOverrides >= numNegInds, E_FRACTIONAL_EXPONENT_WITHOUT_NEG_OVERRIDE, caller);
862
863 int allNegsOverriden;
864
865 // if there are 16 or fewer qubits (0.5mB cache), use a stack array to tick off overrides
866 if (numQubits < 16) {
867
868 // flags for {-1,-2,...}; at index {abs(-1)-1, abs(-2)-2, ...}
869 long long int negIsOverriden[32768]; // [numNegInds];
870 for (int i=0; i<numNegInds; i++)
871 negIsOverriden[i] = 0;
872
873 for (int v=0; v<numOverrides; v++)
874 if (overrideInds[v] < 0)
875 negIsOverriden[ -1 - overrideInds[v] ] = 1;
876
877 allNegsOverriden = 1;
878 for (int i=0; i<numNegInds; i++) {
879 if (!negIsOverriden[i]) {
880 allNegsOverriden = 0;
881 break;
882 }
883 }
884 }
885 // otherwise, we must trust the user, else impose significant slowdowns on good users
886 else {
887 allNegsOverriden = 1;
888 }
